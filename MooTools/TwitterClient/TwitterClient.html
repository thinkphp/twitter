MooTools TwitterClient Plugin


window.addEvent('domready',function(){
             new TwitterClient('codepo8',{
                 count: 5,
                 onComplete: function(tweets, user) {
                      $('tweets').set('text','');
                      tweets.each(function(tweet,index){
                        new Element('div',{
                            html: '<img src="'+user.profile_image_url+'" align="left" alt="photo_profile" />; <strong>'+ user.name +'</strong><br/>'+
                            tweet.text + '<br/><span>' + tweet.created_at + ' via ' + tweet.source.replace('\\','') + '</span>',
                            'class': 'tweet clear'
                        }).inject('tweets');
                      });  
                 },
                 onRequest: function() {
                      $('tweets').set('text','Loading...'); 
                 }
             }).retrieve();
});
