Request.TwitterSearch extends Request.JSONP


window.addEvent('domready',function(){
       new Request.TwitterSearch('mootools',{
           data: {
              rpp: 7
           },
           onSuccess: function(tweets){
                 tweets.each(function(tweet){
                      new Element('div',{
                          'class': 'tweet',
                          'html': tweet.text
                       }).inject('tweets','bottom');   
                 });
           }
       }).send();
});
